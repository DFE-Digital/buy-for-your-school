def call
  task = Task.new(
    section: section,
    title: contentful_task.title,
    contentful_id: contentful_task.id,
    order: order,
  )
  begin
    task.save!
    task
  # TODO: why might the record be invalid? we are missing coverage here
  rescue ActiveRecord::RecordInvalid
    raise UnexpectedContentfulModel
  end
end
