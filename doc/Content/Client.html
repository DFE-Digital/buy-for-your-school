def initialize(type = :delivery)
  @client =
    Content::Connector.instance(
      ENV["CONTENTFUL_SPACE"],
      ENV["CONTENTFUL_DELIVERY_TOKEN"],
      ENV["CONTENTFUL_PREVIEW_TOKEN"],
    ).client(type)
end
