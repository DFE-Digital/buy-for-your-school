def call(draft: nil)
  is_draft = draft.nil? ? !journey.all_tasks_completed? : draft

  template = LiquidParser.new(
    template: journey.category.liquid_template,
    answers: answers,
  ).call

  template.prepend("#{draft_msg}\n\n") if is_draft

  document_formatter(template)
end
