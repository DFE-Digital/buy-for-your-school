def call(params:)
  result = Result.new(false, answer)

  safe_params = StringSanitiser.new(args: params.to_hash).call
  answer.assign_attributes(safe_params)

  if answer.valid?
    answer.save!
    answer.step.journey.start!
    ToggleAdditionalSteps.new(step: answer.step).call
    result.success = true
  end

  result
end
